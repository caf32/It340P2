Java "Area Server" 3/6/2016

This java program is a server program that takes as an input radius or a circle, calculates the area and sends it to the requester.


//
// A server that calculates and returns the
// area of a circle whose radius is sent by
// a client.
//

import java.util.*;
import java.net.*; // needed for sockets
import java.io.*;

public class areaserver
{

public static void main(String[] args)
{

ServerSocket serverSocket;
Socket socket;
double radius;
double area;

try
{
// create a server socket

serverSocket = new ServerSocket(4444);

// Listen for a connection request from a client

socket = serverSocket.accept();

// Establish the input and output streams on the socket

PrintWriter out = new
PrintWriter(socket.getOutputStream(), true);
Scanner in = new Scanner(new
InputStreamReader(socket.getInputStream()));

// Wait for the client to send the radius. When it's
// received, calculate the area and send it back to the
// client

radius = in.nextDouble();

area = Math.PI * radius * radius;

out.println(area);

// disconnect: close streams and sockets

out.close();
in.close();
socket.close();
serverSocket.close();

}
catch(IOException e)
{
System.out.println(e);
}

} // end main

} // end areaserver
